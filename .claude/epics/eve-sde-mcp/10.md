## Technical Details

Market group tree walk:
```rust
let mut path = vec![];
let mut current_id = market_group_id;
let mut seen = HashSet::new();
loop {
    if !seen.insert(current_id) { return Err(anyhow!("cycle detected")); }
    let record = fetch_by_id(&store.market_groups, current_id)?;
    path.push(record.clone());
    match record["parentGroupID"].as_u64() {
        Some(parent) => current_id = parent,
        None => break,
    }
    if path.len() > 20 { return Err(anyhow!("exceeded max depth")); }
}
path.reverse(); // root first
```
